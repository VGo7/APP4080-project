What is Good Variable Naming?

Variable names should be descriptive and meaningful, making code self-documenting. Avoid single letters or vague abbreviations that make code harder to understand and maintain.

Beginner
Bad Practice
// Poor variable naming
let x = 5;
let y = 10;
let z = x + y;

function calc(a, b) {
  return a * b;
}

let data = calc(x, y);
Good Practice
// Descriptive variable naming
let width = 5;
let height = 10;
let area = width + height;

function calculateRectangleArea(width, height) {
  return width * height;
}

let rectangleArea = calculateRectangleArea(width, height);

How Should Functions Be Structured?

Functions should follow the Single Responsibility Principle, doing one thing well. They should be small, focused, and have clear, descriptive names that explain their purpose.

Intermediate
Bad Practice
// Function doing too many things
function processUser(user) {
  // Validate
  if (!user.email) return false;
  
  // Transform
  user.email = user.email.toLowerCase();
  
  // Save to database
  db.save(user);
  
  // Send email
  emailService.send(user.email);
  
  // Log
  console.log('User processed');
  
  return true;
}
Good Practice
// Separate concerns into focused functions
function validateUser(user) {
  return user.email !== undefined;
}

function normalizeEmail(email) {
  return email.toLowerCase();
}

function saveUser(user) {
  return db.save(user);
}

function sendWelcomeEmail(email) {
  return emailService.send(email);
}

What is Proper Error Handling?

Error handling should be specific and informative. Catch specific exceptions, provide meaningful error messages, and handle errors gracefully rather than silently failing or crashing.

Intermediate
Bad Practice
// Generic error handling
try {
  let data = JSON.parse(response);
  processData(data);
  saveToDatabase(data);
} catch (e) {
  console.log('Error');
}

// Silent failure
function getUserData(id) {
  try {
    return api.getUser(id);
  } catch {
    return null;
  }
}
Good Practice
// Specific error handling
try {
  let data = JSON.parse(response);
  processData(data);
  saveToDatabase(data);
} catch (error) {
  if (error instanceof SyntaxError) {
    console.error('Invalid JSON:', error.message);
  } else if (error instanceof DatabaseError) {
    console.error('Database error:', error.message);
  } else {
    throw error;
  }
}

How Should Code Be Commented?

Comments should explain why, not what. The code itself should be clear enough to show what it does. Use comments to explain complex logic, business rules, or non-obvious decisions.

Beginner
Bad Practice
// Redundant comments
// Add 1 to counter
counter = counter + 1;

// Loop through array
for (let i = 0; i < arr.length; i++) {
  // Print element
  console.log(arr[i]);
}

// Check if x equals 5
if (x == 5) {
  // Do something
  doSomething();
}
Good Practice
// Meaningful comments explaining why
// Increment counter for rate limiting
// API allows max 100 requests per minute
counter++;

// Using for loop instead of forEach
// for better performance with large datasets
for (let i = 0; i < arr.length; i++) {
  console.log(arr[i]);
}

// Magic number from legacy API specification
if (statusCode === 5) {
  handleLegacyResponse();
}

Why Use Constants?

Constants improve code maintainability by centralizing values that are used throughout your application. They prevent magic numbers and make code more readable.

Beginner
Bad Practice
// Hard-coded values everywhere
if (user.age >= 18) {
  allowEntry();
}

if (discount > 0.5) {
  rejectDiscount();
}

setTimeout(checkStatus, 5000);

if (items.length > 100) {
  showWarning();
}
Good Practice
// Using named constants
const MINIMUM_AGE = 18;
const MAX_DISCOUNT_RATE = 0.5;
const STATUS_CHECK_INTERVAL = 5000;
const MAX_ITEMS_PER_PAGE = 100;

if (user.age >= MINIMUM_AGE) {
  allowEntry();
}

if (discount > MAX_DISCOUNT_RATE) {
  rejectDiscount();
}

setTimeout(checkStatus, STATUS_CHECK_INTERVAL);

if (items.length > MAX_ITEMS_PER_PAGE) {
  showWarning();
}

How to Handle Asynchronous Code?

Modern JavaScript provides async/await syntax that makes asynchronous code more readable and easier to maintain than nested callbacks or promise chains.

Intermediate
Bad Practice
// Callback hell
function getUserData(userId) {
  fetchUser(userId, function(error, user) {
    if (error) {
      console.log(error);
    } else {
      fetchPosts(user.id, function(error, posts) {
        if (error) {
          console.log(error);
        } else {
          fetchComments(posts[0].id, function(error, comments) {
            if (error) {
              console.log(error);
            } else {
              displayData(user, posts, comments);
            }
          });
        }
      });
    }
  });
}
Good Practice
// Clean async/await
async function getUserData(userId) {
  try {
    const user = await fetchUser(userId);
    const posts = await fetchPosts(user.id);
    const comments = await fetchComments(posts[0].id);
    
    displayData(user, posts, comments);
  } catch (error) {
    console.error('Error fetching user data:', error);
  }
}

What Are Modern Array Methods?

JavaScript provides powerful built-in array methods that make code more declarative and easier to read compared to traditional loops.

Beginner
Bad Practice
// Using traditional loops
let activeUsers = [];
for (let i = 0; i < users.length; i++) {
  if (users[i].active) {
    activeUsers.push(users[i]);
  }
}

let userNames = [];
for (let i = 0; i < activeUsers.length; i++) {
  userNames.push(activeUsers[i].name);
}

let total = 0;
for (let i = 0; i < prices.length; i++) {
  total += prices[i];
}
Good Practice
// Using modern array methods
const activeUsers = users.filter(user => user.active);

const userNames = activeUsers.map(user => user.name);

const total = prices.reduce((sum, price) => sum + price, 0);

// Or chain them together
const activeUserNames = users
  .filter(user => user.active)
  .map(user => user.name);

Why Use Design Patterns?

Design patterns provide tested solutions to common problems. The Strategy pattern, for example, allows you to define a family of algorithms and make them interchangeable.

Advanced
Bad Practice
// Hardcoded logic with many conditionals
function calculateShipping(order, method) {
  let cost = 0;
  
  if (method === 'standard') {
    cost = order.weight * 0.5;
    if (order.total > 50) cost = 0;
  } else if (method === 'express') {
    cost = order.weight * 1.5 + 10;
    if (order.total > 100) cost *= 0.8;
  } else if (method === 'overnight') {
    cost = order.weight * 3 + 20;
  } else if (method === 'international') {
    cost = order.weight * 5 + 30;
    if (order.destination === 'Canada') cost *= 0.9;
  }
  
  return cost;
}
Good Practice
// Strategy Pattern
class StandardShipping {
  calculate(order) {
    const cost = order.weight * 0.5;
    return order.total > 50 ? 0 : cost;
  }
}

class ExpressShipping {
  calculate(order) {
    let cost = order.weight * 1.5 + 10;
    return order.total > 100 ? cost * 0.8 : cost;
  }
}

class ShippingCalculator {
  constructor(strategy) {
    this.strategy = strategy;
  }
  
  calculate(order) {
    return this.strategy.calculate(order);
  }
}

// Usage
const calculator = new ShippingCalculator(new ExpressShipping());
const cost = calculator.calculate(order);

What is Object Destructuring?

Destructuring allows you to extract properties from objects and arrays into distinct variables, making code cleaner and more readable.

Intermediate
Bad Practice
// Repetitive property access
function displayUser(user) {
  console.log(user.name);
  console.log(user.email);
  console.log(user.age);
  
  const street = user.address.street;
  const city = user.address.city;
  const country = user.address.country;
  
  return user.name + ' from ' + city;
}

function createUser(config) {
  const name = config.name;
  const email = config.email;
  const role = config.role || 'user';
  const active = config.active !== undefined ? config.active : true;
  
  return { name, email, role, active };
}
Good Practice
// Using destructuring
function displayUser(user) {
  const { name, email, age, address } = user;
  const { street, city, country } = address;
  
  console.log(name);
  console.log(email);
  console.log(age);
  
  return `${name} from ${city}`;
}

function createUser({ name, email, role = 'user', active = true }) {
  return { name, email, role, active };
}

// Also works with arrays
const [first, second, ...rest] = numbers;
const [, , third] = items; // Skip first two

How to Handle Null and Undefined Values?

Modern JavaScript provides safer ways to handle null and undefined values using optional chaining and nullish coalescing operators.

Beginner
Bad Practice
// Verbose null checking
let userName;
if (user && user.profile && user.profile.name) {
  userName = user.profile.name;
} else {
  userName = 'Guest';
}

let count;
if (data.count !== null && data.count !== undefined) {
  count = data.count;
} else {
  count = 0;
}

// Checking before method call
let upperName;
if (user && user.getName) {
  upperName = user.getName().toUpperCase();
}
Good Practice
// Optional chaining and nullish coalescing
const userName = user?.profile?.name ?? 'Guest';

const count = data.count ?? 0;

// Safe method call
const upperName = user?.getName()?.toUpperCase();

// With default values
const settings = {
  theme: config?.theme ?? 'light',
  notifications: config?.notifications ?? true,
  language: config?.language ?? 'en'
};

How to Prevent Memory Leaks?

Memory leaks occur when references to objects are maintained unnecessarily. Proper cleanup of event listeners, timers, and subscriptions is crucial.

Advanced
Bad Practice
// Memory leaks from uncleaned resources
class DataFetcher {
  constructor() {
    this.timer = setInterval(() => {
      this.fetchData();
    }, 5000);
    
    document.addEventListener('click', this.handleClick);
    
    this.cache = [];
  }
  
  handleClick() {
    // 'this' context lost
    console.log(this.cache);
  }
  
  fetchData() {
    fetch('/api/data')
      .then(res => res.json())
      .then(data => {
        this.cache.push(data); // Unbounded growth
      });
  }
}

// No cleanup when done!
Good Practice
// Proper resource management
class DataFetcher {
  constructor() {
    this.timer = setInterval(() => {
      this.fetchData();
    }, 5000);
    
    this.handleClick = this.handleClick.bind(this);
    document.addEventListener('click', this.handleClick);
    
    this.cache = new Map();
    this.maxCacheSize = 100;
  }
  
  handleClick() {
    console.log(this.cache);
  }
  
  fetchData() {
    fetch('/api/data')
      .then(res => res.json())
      .then(data => {
        if (this.cache.size >= this.maxCacheSize) {
          const firstKey = this.cache.keys().next().value;
          this.cache.delete(firstKey);
        }
        this.cache.set(data.id, data);
      });
  }
  
  destroy() {
    clearInterval(this.timer);
    document.removeEventListener('click', this.handleClick);
    this.cache.clear();
  }
}

Why Use Template Literals?

Template literals provide a cleaner syntax for string interpolation and multi-line strings compared to string concatenation.

Beginner
Bad Practice
// String concatenation
const greeting = 'Hello, ' + user.name + '!';

const message = 'You have ' + unreadCount + ' unread messages.';

const url = baseUrl + '/api/' + version + '/users/' + userId;

const html = '
' +
  '
' + title + '
' +
  '
' + description + '
' +
  '
';

const multiLine = 'This is line one.\n' +
  'This is line two.\n' +
  'This is line three.';
Good Practice
// Template literals
const greeting = `Hello, ${user.name}!`;

const message = `You have ${unreadCount} unread messages.`;

const url = `${baseUrl}/api/${version}/users/${userId}`;

const html = `
  

    
${title}

    
${description}

  

`;

const multiLine = `
  This is line one.
  This is line two.
  This is line three.
`;
